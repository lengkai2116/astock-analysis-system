## 二、具体优化方案（按优先级排序）

### 优化1：预计算指标缓存（高优先级）

#### 问题
当前实现每次请求都实时计算指标（`TechnicalIndicatorEngine.calculate_all_indicators`），重复计算浪费资源。

#### 借鉴Qlib
Qlib采用**预计算 + 增量更新**机制，将常用指标提前计算并缓存。

#### 实现方案

**新增文件**：`backend/app/data/precompute_indicator_manager.py`
```python
import pandas as pd
from datetime import datetime, timedelta
from app.indicators import TechnicalIndicatorEngine

class PrecomputeIndicatorManager:
    """
    预计算指标管理器
    - 借鉴Qlib的预计算策略
    - 仅在数据更新时增量重算
    """
    
    def __init__(self, cache_manager):
        self.cache_manager = cache_manager
        self.engine = TechnicalIndicatorEngine()
    
    def precompute_all_indicators(self, ts_code, force=False):
        """
        预计算所有指标
        优化点：仅重算新增数据的指标
        """
        # 获取全部数据
        df = self._get_daily_data(ts_code)
        if len(df) < 30:
            return
        
        # 检查已有缓存时间，仅增量计算
        latest_cached = self._get_latest_cached_date(ts_code)
        if latest_cached and not force:
            # 只处理新增数据
            new_data = df[df['trade_date'] > latest_cached]
            if len(new_data) < 3:  # 数据太少，不值得增量算
                return
            # 确保有足够历史计算指标
            start_idx = max(0, df.index.get_loc(new_data.index[0]) - 60)
            df_to_calc = df.iloc[start_idx:]
        else:
            df_to_calc = df
        
        # 计算指标
        result = self.engine.calculate_all_indicators(df_to_calc)
        
        # 批量缓存所有指标值
        self._batch_cache_indicators(result, ts_code)
    
    def _batch_cache_indicators(self, df, ts_code):
        """
        批量缓存指标（一次写入多条）
        优化：避免逐条INSERT，性能提升10-100倍
        """
        indicator_cols = [
            'ma5', 'ma10', 'ma20', 
            'macd_dif', 'macd_dea', 'macd_hist',
            'rsi14', 
            'kdj_k', 'kdj_d', 'kdj_j',
            'boll_upper', 'boll_mid', 'boll_lower',
            'vol_ma5', 'vol_ma10'
        ]
        
        records = []
        for _, row in df.iterrows():
            for col in indicator_cols:
                val = row.get(col)
                if pd.notna(val):
                    records.append({
                        'ts_code': ts_code,
                        'trade_date': row['trade_date'],
                        'indicator_name': col,
                        'value': float(val),
                        'cached_at': datetime.now()
                    })
        
        if records:
            # 批量写入DuckDB
            self.cache_manager.batch_cache_indicators(records)
    
    def get_precomputed_indicators(self, ts_code):
        """
        获取预计算指标
        优先从缓存获取，避免实时计算
        """
        result = pd.DataFrame()
        for indicator in ['ma5', 'ma10', 'ma20', 'macd_dif', 'rsi14']:  # 常用指标
            df = self.cache_manager.get_indicator_data(ts_code, indicator)
            if not df.empty:
                if result.empty:
                    result = df[['trade_date', 'value']].rename(columns={'value': indicator})
                else:
                    result = result.merge(df[['trade_date', 'value']], 
                                        on='trade_date', 
                                        how='outer',
                                        suffixes=('', f'_{indicator}'))
                    result = result.rename(columns={'value': indicator})
        return result
```

def batch_cache_indicators(self, records):
    """
    批量缓存指标（高性能写入）
    借鉴Qlib的批量处理理念
    """
    if not records:
        return

        SELECT ts_code, trade_date, indicator_name, value, cached_at
        FROM temp_indicators
    """)
    self.conn.commit()

### 优化2：增量更新指标（高优先级）

#### 问题
每次添加新数据时，所有历史指标都重算，浪费算力。

#### 借鉴Qlib
Qlib采用**滑动窗口增量更新**，只重新计算窗口内变化的部分。

#### 实现方案

**修改**：`backend/app/data/precompute_indicator_manager.py`
```python
def incremental_update_indicators(self, ts_code, new_data):
    """
    增量更新指标
    - 仅重算受新数据影响的指标窗口
    - 借鉴Qlib的高效更新策略
    """
    if len(new_data) < 1:
        return
    
    # 获取历史数据（包括足够的上下文）
    old_df = self._get_daily_data(ts_code)
    # 留足窗口大小（最长MA20，MACD26）
    window_size = 30
    
    if len(old_df) > 0:
        # 合并新老数据
        combined = pd.concat([old_df, new_data]).drop_duplicates('trade_date').sort_values('trade_date')
        start_idx = max(0, len(combined) - len(new_data) - window_size)
        df_to_recalc = combined.iloc[start_idx:]
        
        # 重算指标
        with_indicators = self.engine.calculate_all_indicators(df_to_recalc)
        
        # 只更新增量产生新值的部分
        new_dates = [d.strftime('%Y-%m-%d') for d in new_data['trade_date']] if isinstance(new_data, pd.DataFrame) else new_data
        update_df = with_indicators[with_indicators['trade_date'].isin(new_dates)]
        self._batch_cache_indicators(update_df, ts_code)
```

### 优化3：DuckDB配置与性能调优（中优先级）

#### 问题
当前DuckDB配置保守（`threads: 1`），没有充分利用多核资源。

#### 借鉴Qlib
Qlib使用列式存储、向量化计算和多线程充分榨干性能。

#### 实现方案

**修改**：`backend/app/data/enhanced_cache_manager.py`
```python
def __init__(self):
    self.redis_cache = RedisCacheManager()
    
    # DuckDB优化配置 - 借鉴Qlib的高性能设置
    data_dir = os.getenv('DATA_DIR', '/data')
    self.db_path = os.path.join(data_dir, 'duckdb', 'stock_cache.db')
    os.makedirs(os.path.dirname(self.db_path), exist_ok=True)
    
    # 性能优化配置
    duckdb_config = {
        'threads': os.cpu_count() or 4,  # 使用全部CPU核心
        'memory_limit': '4GB',  # 加大内存使用限制
        'enable_external_access': 'false',
        'max_memory': '4GB',
        'temp_directory': os.path.join(data_dir, 'duckdb', 'temp')
    }
    
    os.makedirs(duckdb_config['temp_directory'], exist_ok=True)
    
    try:
        self.conn = duckdb.connect(self.db_path, config=duckdb_config)
        # 启用列式扫描优化
        self.conn.execute("PRAGMA enable_object_cache")
        # 设置分析模式
        self.conn.execute("PRAGMA force_index_scan")
    except Exception as e:
        print(f"⚠️ 降级配置: {e}")
        self.conn = duckdb.connect(self.db_path, config={'threads': 2})
```

**性能优化点**：
- ✅ 使用所有CPU核心（从1提升到4或更多）
- ✅ 增加内存限制（默认保守值）
- ✅ 启用对象缓存减少重复IO
- ✅ 使用临时目录优化溢出处理

### 优化4：预加载和预热热点数据（中优先级）

#### 问题
冷启动或首次查询某只股票时，需要从磁盘加载，响应缓慢。

#### 借鉴Qlib
Qlib在启动时预加载活跃交易股票的数据到内存。

#### 实现方案

**新增文件**：`backend/app/data/preloader.py`
```python
import threading
import time
from datetime import datetime, timedelta

class DataPreloader:
    """
    数据预加载器
    - 借鉴Qlib的预热策略
    - 后台预热自选股、热门股
    """
    
    def __init__(self, cache_manager, market_service):
        self.cache_manager = cache_manager
        self.market_service = market_service
        self.is_running = False
        self.preload_thread = None
        
        # 预加载股票列表 - 自选股、沪深300、热门股
        self.preload_symbols = []
    
    def start(self):
        """启动后台预加载"""
        if self.is_running:
            return
        
        self.is_running = True
        self.preload_thread = threading.Thread(target=self._preload_loop, daemon=True)
        self.preload_thread.start()
        print("🚀 数据预加载服务已启动")
    
    def _preload_loop(self):
        """预热循环"""
        while self.is_running:
            try:
                self._preload_hot_stocks()
                self._precompute_hot_indicators()
            except Exception as e:
                print(f"⚠️ 预加载异常: {e}")
            
            # 每小时检查更新一次
            time.sleep(3600)
    
    def _preload_hot_stocks(self):
        """预加载热门股票数据到内存缓存"""
        # 获取热门股（示例）
        hot_stocks = ['600519.SH', '000001.SZ', '000002.SZ', 
                      '601318.SH', '601398.SH']
        
        for symbol in hot_stocks:
            try:
                # 先从DuckDB缓存，再加载到Redis
                df = self.cache_manager.get_cached_daily(symbol)
                if not df.empty:
                    # 缓存最近250天数据到Redis
                    self.cache_manager.redis_cache.set_daily_data(symbol, df.tail(250))
            except Exception:
                pass
    
    def _precompute_hot_indicators(self):
        """预计算热门股指标"""
        for symbol in self.preload_symbols:
            try:
                from app.data.precompute_indicator_manager import PrecomputeIndicatorManager
                precompute_manager = PrecomputeIndicatorManager(self.cache_manager)
                precompute_manager.precompute_all_indicators(symbol)
            except Exception:
                pass
```

### 优化5：批量操作与向量化（中优先级）

#### 问题
现有代码许多地方是逐行处理，效率不高。

#### 借鉴Qlib
Qlib大量使用Pandas/Numpy向量化，避免Python层循环。

#### 实现方案 - 优化指标计算

**优化**：`backend/app/indicators/__init__.py`
```python
def calculate_all_indicators(self, df: pd.DataFrame) -> pd.DataFrame:
    """
    优化版计算所有技术指标
    - 避免多次DataFrame拷贝
    - 链式调用减少中间步骤
    - 向量化计算（避免逐行）
    """
    # 直接在原DataFrame上计算，避免拷贝
    if len(df) < 20:
        return df
    
    # 一次性设置所有指标
    result = df.copy()
    
    # 所有计算直接向量化
    close = result['close'].values
    high = result['high'].values
    low = result['low'].values
    vol = result['vol'].values
    
    n = len(result)
    
    # 1. MA计算 - 直接用pd.rolling
    if n >= 5:
        result['ma5'] = result['close'].rolling(5).mean()
    if n >= 10:
        result['ma10'] = result['close'].rolling(10).mean()
    if n >= 20:
        result['ma20'] = result['close'].rolling(20).mean()
    
    # 2. MACD计算 - 直接用numpy向量化
    if n >= 26:
        # 优化：避免多次Series转换
        ema12 = pd.Series(close).ewm(span=12, adjust=False).mean().values
        ema26 = pd.Series(close).ewm(span=26, adjust=False).mean().values
        result['macd_dif'] = ema12 - ema26
        result['macd_dea'] = pd.Series(result['macd_dif']).ewm(span=9, adjust=False).mean().values
        result['macd_hist'] = 2 * (result['macd_dif'] - result['macd_dea'])
    
    # ... 其他指标同理优化
    
    return result
```
