## 二、色彩系统修正（要求1）

### 2.1 A股色彩标准

**中国A股的色彩约定**：
- **红色 = 上涨/买入/利好**
- **绿色 = 下跌/卖出/利空**

### 2.2 修正后的色彩系统

<script>
export default {
  name: 'StrategySelector',
  data() {
    return {
      strategies: [...strategyLibrary]
    }
  },
  methods: {
    toggleStrategy(strategy) {
      strategy.enabled = !strategy.enabled
      this.$emit('strategy-change', this.strategies.filter(s => s.enabled))
    },
    updateParam(strategy, key, value) {
      strategy.params[key] = value
      this.$emit('strategy-change', this.strategies.filter(s => s.enabled))
    },
    formatParamLabel(key) {
      const labels = {
        period: '周期',
        fast: '快线',
        slow: '慢线',
        signal: '信号线',
        stdDev: '标准差',
        threshold: '阈值',
        lookback: '回望'
      }
      return labels[key] || key
    }
  }
}
</script>

### 3.4 策略建议输出设计

#### 3.4.1 策略建议数据结构

```javascript
// 单个策略建议
interface StrategyAdvice {
  strategyId: string;
  strategyName: string;
  timestamp: number;

  signalType: 'BUY' | 'SELL' | 'HOLD' | 'WATCH';
  confidence: number;

  priceTarget?: number;
  stopLoss?: number;
  positionSize?: number;

  reasons: string[];
  keyIndicators: {
    name: string;
    value: any;
    direction: 'UP' | 'DOWN' | 'NEUTRAL';
  }[];

  chartMarkers?: {
    type: 'POINT' | 'LINE' | 'AREA';
    time: number;
    price?: number;
    label?: string;
    color?: string;
  }[];
}

// 多策略综合报告
interface CompositeReport {
  stock: {
    symbol: string;
    name: string;
    currentPrice: number;
  };

  strategies: StrategyAdvice[];

  consensus: {
    signalCount: {
      BUY: number;
      SELL: number;
      HOLD: number;
      WATCH: number;
    };
    recommendation: string;
    suggestedPosition: number;
    keyRisks: string[];
  };

  generatedAt: number;
}
```

<script>
export default {
  name: 'StrategyAdvicePanel',
  props: {
    symbol: { type: String, required: true },
    activeStrategies: { type: Array, default: () => [] }
  },
  data() {
    return {
      loading: true,
      advices: [],
      composite: null
    }
  },
  computed: {
    consensus() {
      if (!this.composite) return '分析中...'
      const { signalCount } = this.composite
      if (signalCount.BUY > signalCount.SELL) return '看多'
      if (signalCount.SELL > signalCount.BUY) return '看空'
      return '震荡'
    },
    consensusColor() {
      if (!this.composite) return 'default'
      const { signalCount } = this.composite
      if (signalCount.BUY > signalCount.SELL) return 'red'
      if (signalCount.SELL > signalCount.BUY) return 'green'
      return 'orange'
    }
  },
  watch: {
    symbol: {
      immediate: true,
      handler() {
        this.runAnalysis()
      }
    },
    activeStrategies: {
      deep: true,
      handler() {
        this.runAnalysis()
      }
    }
  },
  methods: {
    async runAnalysis() {
      this.loading = true
      try {
        const response = await this.$http.post('/api/strategies/analyze', {
          symbol: this.symbol,
          strategies: this.activeStrategies.map(s => ({
            id: s.id,
            params: s.params
          }))
        })

        this.advices = response.advices
        this.composite = response.composite
        this.$emit('advice-update', response)
      } catch (error) {
        console.error('Analysis failed:', error)
      } finally {
        this.loading = false
      }
    },
    getSignalColor(type) {
      const colors = {
        BUY: '#EF4444',
        SELL: '#22C55E',
        HOLD: '#F59E0B',
        WATCH: '#64748B'
      }
      return colors[type] || '#64748B'
    },
    getSignalLabel(type) {
      const labels = {
        BUY: '买入',
        SELL: '卖出',
        HOLD: '持有',
        WATCH: '观望'
      }
      return labels[type] || type
    },
    getIndicatorClass(direction) {
      return {
        UP: 'up',
        DOWN: 'down',
        NEUTRAL: ''
      }[direction]
    },
    formatIndicatorValue(ind) {
      if (typeof ind.value === 'number') {
        return ind.value.toFixed(2)
      }
      return String(ind.value)
    },
    exportReport() {
      this.$emit('export-report', {
        advices: this.advices,
        composite: this.composite
      })
    }
  }
}
</script>

<script>
export default {
  name: 'AIAnalysisPage',
  data() {
    return {
      selectedSymbol: '',
      stockList: [
        { symbol: '600519.SH', name: '贵州茅台' },
        { symbol: '000001.SZ', name: '平安银行' },
        { symbol: '600000.SH', name: '浦发银行' }
      ],
      analyzing: false,
      sessionId: null,
      progress: 0,
      progressText: '准备中...',
      analysisLog: [],
      roles: [
        { id: 'tech', name: '技术分析师', icon: '📈', status: 'waiting' },
        { id: 'fundamental', name: '基本面分析师', icon: '📊', status: 'waiting' },
        { id: 'macro', name: '宏观策略师', icon: '🌐', status: 'waiting' },
        { id: 'risk', name: '风险控制官', icon: '⚠️', status: 'waiting' },
        { id: 'manager', name: '资深基金经理', icon: '👨‍💼', status: 'waiting' }
      ],
      finalReport: null
    }
  },
  methods: {
    async startAnalysis() {
      if (!this.selectedSymbol) return

      this.analyzing = true
      this.analysisLog = []
      this.finalReport = null
      this.roles.forEach(r => {
        r.status = 'waiting'
        r.report = null
      })

      this.addLog('', '🚀', `开始分析 ${this.selectedSymbol}...`)

      try {
        const session = await this.$http.post('/api/ai/start', {
          symbol: this.selectedSymbol
        })
        this.sessionId = session.sessionId

        this.pollProgress()
      } catch (error) {
        console.error('Start analysis failed:', error)
        this.analyzing = false
      }
    },
    async pollProgress() {
      const poll = async () => {
        if (!this.analyzing) return

        try {
          const status = await this.$http.get(`/api/ai/progress/${this.sessionId}`)

          this.progress = status.progress
          this.progressText = status.statusText

          if (status.logs) {
            status.logs.forEach(log => {
              if (!this.analysisLog.find(l => l.timestamp === log.timestamp)) {
                this.addLog(log.role, log.icon, log.content)
              }
            })
          }

          if (status.roleStatus) {
            status.roleStatus.forEach(roleStatus => {
              const role = this.roles.find(r => r.id === roleStatus.roleId)
              if (role) {
                role.status = roleStatus.status
                role.report = roleStatus.report
              }
            })
          }

          if (status.finalReport) {
            this.finalReport = status.finalReport
            this.analyzing = false
            return
          }

          setTimeout(poll, 1000)
        } catch (error) {
          console.error('Poll error:', error)
        }
      }
      poll()
    },
    stopAnalysis() {
      this.analyzing = false
      if (this.sessionId) {
        this.$http.post(`/api/ai/cancel/${this.sessionId}`)
      }
    },
    addLog(role, icon, content) {
      this.analysisLog.push({
        role,
        icon,
        content,
        timestamp: Date.now(),
        type: role === 'error' ? 'error' : 'normal'
      })
    },
    formatTime(timestamp) {
      const date = new Date(timestamp)
      return `${date.getHours().toString().padStart(2, '0')}:${date.getMinutes().toString().padStart(2, '0')}:${date.getSeconds().toString().padStart(2, '0')}`
    },
    getRatingLabel(rating) {
      const labels = {
        STRONGLY_BUY: '强烈看多',
        BUY: '看多',
        HOLD: '中性',
        SELL: '看空',
        STRONGLY_SELL: '强烈看空'
      }
      return labels[rating] || rating
    },
    onSymbolChange() {
      this.sessionId = null
      this.finalReport = null
      this.analysisLog = []
    }
  }
}
