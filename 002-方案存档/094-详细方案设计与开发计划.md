```python
from app import db
from datetime import datetime
from sqlalchemy import JSON, Enum
import enum

class StrategySignal(enum.Enum):
    BULLISH = "bullish"      # 看多
    BEARISH = "bearish"      # 看空
    NEUTRAL = "neutral"      # 中性
    WATCH = "watch"          # 观察

class StrategyTemplateType(enum.Enum):
    INDICATOR = "indicator"    # 指标型
    SELECTION = "selection"    # 选股型
    PORTFOLIO = "portfolio"    # 组合型

class StrategyOutput(db.Model):
    """策略输出核心模型"""
    __tablename__ = 'strategy_outputs'
    
    id = db.Column(db.Integer, primary_key=True)
    ts_code = db.Column(db.String(10), nullable=False)
    strategy_name = db.Column(db.String(100), nullable=False)
    signal = db.Column(db.Enum(StrategySignal), nullable=False)
    confidence = db.Column(db.DECIMAL(5, 2))  # 置信度 0-1
    entry_zone_low = db.Column(db.DECIMAL(10, 2))
    entry_zone_high = db.Column(db.DECIMAL(10, 2))
    risk_line = db.Column(db.DECIMAL(10, 2))
    target_zone_low = db.Column(db.DECIMAL(10, 2))
    target_zone_high = db.Column(db.DECIMAL(10, 2))
    position_suggestion = db.Column(db.String(50))  # 建议仓位，如"10%-20%"
    holding_period = db.Column(db.String(50))       # 建议持有期，如"5-20个交易日"
    evidence = db.Column(JSON)                      # 证据链数组
    risk_notes = db.Column(JSON)                    # 风险提示数组
    signal_date = db.Column(db.Date, nullable=False)
    created_at = db.Column(db.DateTime, default=datetime.now)
    
    def to_dict(self):
        return {
            'id': self.id,
            'ts_code': self.ts_code,
            'strategy_name': self.strategy_name,
            'signal': self.signal.value if self.signal else None,
            'confidence': float(self.confidence) if self.confidence else None,
            'entry_zone': [
                float(self.entry_zone_low) if self.entry_zone_low else None,
                float(self.entry_zone_high) if self.entry_zone_high else None
            ] if (self.entry_zone_low or self.entry_zone_high) else None,
            'risk_line': float(self.risk_line) if self.risk_line else None,
            'target_zone': [
                float(self.target_zone_low) if self.target_zone_low else None,
                float(self.target_zone_high) if self.target_zone_high else None
            ] if (self.target_zone_low or self.target_zone_high) else None,
            'position_suggestion': self.position_suggestion,
            'holding_period': self.holding_period,
            'evidence': self.evidence,
            'risk_notes': self.risk_notes,
            'signal_date': self.signal_date.strftime('%Y-%m-%d') if self.signal_date else None,
            'created_at': self.created_at.strftime('%Y-%m-%d %H:%M:%S')
        }
```

### 2.3 策略输出协议服务

**文件**: `backend/app/services/strategy_output_service.py` (新建)

```python
from typing import Dict, List, Optional
from datetime import date
from app.models.strategy import (
    StrategyOutput, StrategySignal, StrategyTemplateV2
)
from app import db

class StrategyOutputService:
    """策略输出服务"""
    
    @staticmethod
    def create_strategy_output(
        ts_code: str,
        strategy_name: str,
        signal: str,
        signal_date: date,
        confidence: Optional[float] = None,
        entry_zone: Optional[List[float]] = None,
        risk_line: Optional[float] = None,
        target_zone: Optional[List[float]] = None,
        position_suggestion: Optional[str] = None,
        holding_period: Optional[str] = None,
        evidence: Optional[List[str]] = None,
        risk_notes: Optional[List[str]] = None
    ) -> StrategyOutput:
        """创建策略输出"""
        output = StrategyOutput(
            ts_code=ts_code,
            strategy_name=strategy_name,
            signal=StrategySignal(signal),
            signal_date=signal_date,
            confidence=confidence,
            position_suggestion=position_suggestion,
            holding_period=holding_period,
            evidence=evidence or [],
            risk_notes=risk_notes or []
        )
        
        if entry_zone and len(entry_zone) >= 2:
            output.entry_zone_low = entry_zone[0]
            output.entry_zone_high = entry_zone[1]
        
        output.risk_line = risk_line
        
        if target_zone and len(target_zone) >= 2:
            output.target_zone_low = target_zone[0]
            output.target_zone_high = target_zone[1]
        
        db.session.add(output)
        db.session.commit()
        return output
    
    @staticmethod
    def get_strategy_outputs(
        ts_code: Optional[str] = None,
        strategy_name: Optional[str] = None,
        start_date: Optional[date] = None,
        end_date: Optional[date] = None,
        limit: int = 100
    ) -> List[StrategyOutput]:
        """获取策略输出列表"""
        query = StrategyOutput.query
        
        if ts_code:
            query = query.filter_by(ts_code=ts_code)
        if strategy_name:
            query = query.filter_by(strategy_name=strategy_name)
        if start_date:
            query = query.filter(StrategyOutput.signal_date >= start_date)
        if end_date:
            query = query.filter(StrategyOutput.signal_date <= end_date)
        
        return query.order_by(StrategyOutput.signal_date.desc()).limit(limit).all()
```

### 2.4 策略模板服务

**文件**: `backend/app/services/strategy_template_service.py` (新建)

```python
import re
from typing import Dict, List, Optional
from app import db
from app.models.strategy import StrategyTemplateV2, StrategyTemplateType

class StrategyTemplateService:
    """策略模板服务"""
    
    @staticmethod
    def parse_template_code(code: str) -> Dict:
        """解析模板代码，提取元数据和参数"""
        result = {
            'name': None,
            'description': None,
            'params': []
        }
        
        # 提取名称
        name_match = re.search(r'my_strategy_name\s*=\s*["\']([^"\']+)["\']', code)
        if name_match:
            result['name'] = name_match.group(1)
        
        # 提取描述
        desc_match = re.search(r'my_strategy_description\s*=\s*["\']([^"\']+)["\']', code)
        if desc_match:
            result['description'] = desc_match.group(1)
        
        # 提取参数
        param_pattern = r'#\s*@param\s+(\w+)\s+(\w+)\s+([^\s#]+)\s*(.*)'
        params = re.findall(param_pattern, code)
        
        for name, param_type, default, description in params:
            result['params'].append({
                'name': name,
                'type': param_type,
                'default': default,
                'description': description.strip()
            })
        
        return result
    
    @staticmethod
    def create_template(
        name: str,
        description: str,
        template_type: str,
        code_template: str,
        author: Optional[str] = None,
        is_system: bool = False
    ) -> StrategyTemplateV2:
        """创建策略模板"""
        parsed = StrategyTemplateService.parse_template_code(code_template)
        
        template = StrategyTemplateV2(
            name=name or parsed.get('name', '未命名策略'),
            description=description or parsed.get('description', ''),
            template_type=StrategyTemplateType(template_type),
            code_template=code_template,
            parameters=parsed.get('params', []),
            is_system=is_system,
            author=author
        )
        
        db.session.add(template)
        db.session.commit()
        return template
    
    @staticmethod
    def get_templates(
        template_type: Optional[str] = None,
        is_system: Optional[bool] = None
    ) -> List[StrategyTemplateV2]:
        """获取策略模板列表"""
        query = StrategyTemplateV2.query.filter_by(is_active=True)
        
        if template_type:
            query = query.filter_by(template_type=StrategyTemplateType(template_type))
        if is_system is not None:
            query = query.filter_by(is_system=is_system)
        
        return query.order_by(StrategyTemplateV2.usage_count.desc()).all()
```

### 2.5 新增API路由

**文件**: `backend/app/routes/strategy.py` (新建)

```python
from flask import Blueprint, request, jsonify
from app import db
from app.services.strategy_output_service import StrategyOutputService
from app.services.strategy_template_service import StrategyTemplateService
from datetime import datetime

strategy_bp = Blueprint('strategy', __name__, url_prefix='/api/v2/strategy')

@strategy_bp.route('/outputs', methods=['GET'])
def get_strategy_outputs():
    """获取策略输出列表"""
    ts_code = request.args.get('ts_code')
    strategy_name = request.args.get('strategy_name')
    start_date = request.args.get('start_date')
    end_date = request.args.get('end_date')
    limit = int(request.args.get('limit', 100))
    
    if start_date:
        start_date = datetime.strptime(start_date, '%Y-%m-%d').date()
    if end_date:
        end_date = datetime.strptime(end_date, '%Y-%m-%d').date()
    
    outputs = StrategyOutputService.get_strategy_outputs(
        ts_code=ts_code,
        strategy_name=strategy_name,
        start_date=start_date,
        end_date=end_date,
        limit=limit
    )
    
    return jsonify({
        'success': True,
        'data': [o.to_dict() for o in outputs]
    })

@strategy_bp.route('/templates', methods=['GET'])
def get_templates():
    """获取策略模板列表"""
    template_type = request.args.get('template_type')
    is_system = request.args.get('is_system')
    
    if is_system is not None:
        is_system = is_system.lower() == 'true'
    
    templates = StrategyTemplateService.get_templates(
        template_type=template_type,
        is_system=is_system
    )
    
    return jsonify({
        'success': True,
        'data': [t.to_dict() for t in templates]
    })

@strategy_bp.route('/templates', methods=['POST'])
def create_template():
    """创建策略模板"""
    data = request.json
    template = StrategyTemplateService.create_template(
        name=data.get('name'),
        description=data.get('description'),
        template_type=data.get('template_type', 'indicator'),
        code_template=data.get('code_template'),
        author=data.get('author')
    )
    return jsonify({'success': True, 'data': template.to_dict()})

@strategy_bp.route('/templates/<int:template_id>', methods=['GET'])
def get_template(template_id):
    """获取策略模板详情"""
    from app.models.strategy import StrategyTemplateV2
    template = StrategyTemplateV2.query.get_or_404(template_id)
    return jsonify({'success': True, 'data': template.to_dict()})
```

```python
import pandas as pd
import numpy as np
from typing import Dict, List, Optional, Tuple
from dataclasses import dataclass, field
from datetime import date, datetime, timedelta
from enum import Enum

class BacktestConfig:
    """回测配置"""
    def __init__(self):
        self.initial_capital: float = 100000.0
        self.commission_rate: float = 0.0003  # 佣金费率
        self.stamp_duty_rate: float = 0.001   # 印花税（仅卖出）
        self.slippage_rate: float = 0.001     # 滑点
        self.min_trade_unit: int = 100        # 最小交易单位（手）
        self.t_plus_one: bool = True          # T+1规则
        self.price_limit: float = 0.1         # 涨跌停幅度 10%
        self.max_single_position: float = 0.3 # 单标的最大仓位

class Signal(Enum):
    BUY = "buy"
    SELL = "sell"
    HOLD = "hold"

@dataclass
class Trade:
    """交易记录"""
    trade_date: date
    ts_code: str
    trade_type: str  # buy/sell
    price: float
    quantity: int
    amount: float
    commission: float
    stamp_duty: float
    slippage: float
    reason: str

@dataclass
class Position:
    """持仓记录"""
    ts_code: str
    quantity: int
    avg_cost: float
    available_quantity: int  # T+1可用数量
    last_trade_date: Optional[date] = None

@dataclass
class BacktestResult:
    """回测结果"""
    portfolio_values: pd.Series = field(default_factory=pd.Series)
    benchmark_values: Optional[pd.Series] = None
    trades: List[Trade] = field(default_factory=list)
    positions: Dict[str, Position] = field(default_factory=dict)
    daily_returns: pd.Series = field(default_factory=pd.Series)
    metrics: Dict = field(default_factory=dict)
    signal_records: List[Dict] = field(default_factory=list)

class AShareBacktestEngine:
    """A股回测引擎"""
    
    def __init__(self, config: Optional[BacktestConfig] = None):
        self.config = config or BacktestConfig()
    
    def run_backtest(
        self,
        price_data: pd.DataFrame,
        signals: pd.DataFrame,
        benchmark_data: Optional[pd.DataFrame] = None
    ) -> BacktestResult:
        """
        运行回测
        
        参数:
            price_data: 价格数据，包含ts_code, trade_date, open, high, low, close, pre_close
            signals: 信号数据，包含ts_code, trade_date, signal
            benchmark_data: 基准数据（可选）
        """
        result = BacktestResult()
        cash = self.config.initial_capital
        positions: Dict[str, Position] = {}
        portfolio_values = []
        
        dates = sorted(price_data['trade_date'].unique())
        price_dict = self._build_price_dict(price_data)
        
        for i, current_date in enumerate(dates):
            # 1. 更新可用数量（T+1）
            self._update_available_quantity(positions, current_date)
            
            # 2. 计算当前市值
            current_value = self._calculate_portfolio_value(
                cash, positions, price_dict, current_date
            )
            portfolio_values.append({
                'trade_date': current_date,
                'value': current_value
            })
            
            # 3. 处理当日信号
            day_signals = signals[signals['trade_date'] == current_date]
            for _, signal_row in day_signals.iterrows():
                if signal_row['signal'] == 'buy':
                    cash = self._execute_buy(
                        signal_row, cash, positions, price_dict,
                        current_date, result
                    )
                elif signal_row['signal'] == 'sell':
                    cash = self._execute_sell(
                        signal_row, cash, positions, price_dict,
                        current_date, result
                    )
        
        # 计算最终指标
        result.portfolio_values = pd.Series(
            [v['value'] for v in portfolio_values],
            index=[v['trade_date'] for v in portfolio_values]
        )
        result.positions = positions
        
        if benchmark_data is not None:
            result.benchmark_values = self._normalize_benchmark(
                benchmark_data, self.config.initial_capital
            )
        
        result.metrics = self._calculate_metrics(result.portfolio_values, result.benchmark_values)
        
        return result
    
    def _build_price_dict(self, price_data: pd.DataFrame) -> Dict:
        """构建价格字典"""
        price_dict = {}
        for _, row in price_data.iterrows():
            key = (row['ts_code'], row['trade_date'])
            price_dict[key] = {
                'open': row['open'],
                'high': row['high'],
                'low': row['low'],
                'close': row['close'],
                'pre_close': row.get('pre_close', row['close'])
            }
        return price_dict
    
    def _update_available_quantity(self, positions: Dict[str, Position], current_date: date):
        """更新可用数量（T+1）"""
        for pos in positions.values():
            if pos.last_trade_date and pos.last_trade_date < current_date:
                pos.available_quantity = pos.quantity
    
    def _calculate_portfolio_value(
        self, cash: float, positions: Dict[str, Position],
        price_dict: Dict, current_date: date
    ) -> float:
        """计算组合市值"""
        value = cash
        for ts_code, pos in positions.items():
            key = (ts_code, current_date)
            if key in price_dict:
                price = price_dict[key]['close']
                value += pos.quantity * price * 100  # 1手=100股
        return value
    
    def _execute_buy(
        self, signal_row: Dict, cash: float, positions: Dict[str, Position],
        price_dict: Dict, current_date: date, result: BacktestResult
    ) -> float:
        """执行买入"""
        ts_code = signal_row['ts_code']
        key = (ts_code, current_date)
        
        if key not in price_dict:
            return cash
        
        price_data = price_dict[key]
        pre_close = price_data['pre_close']
        limit_up = pre_close * (1 + self.config.price_limit)
        
        # 使用开盘价或限价
        execute_price = min(price_data['open'], limit_up)
        
        # 计算可买数量
        max_amount = cash * self.config.max_single_position
        quantity = int(max_amount / (execute_price * 100)) * 100  # 100股一手
        
        if quantity < self.config.min_trade_unit * 100:
            return cash
        
        # 计算费用
        amount = execute_price * quantity
        commission = max(amount * self.config.commission_rate, 5)  # 最低5元
        slippage = amount * self.config.slippage_rate
        total_cost = amount + commission + slippage
        
        if total_cost > cash:
            return cash
        
        # 执行买入
        cash -= total_cost
        
        if ts_code in positions:
            pos = positions[ts_code]
            total_shares = pos.quantity + quantity
            total_cost = pos.avg_cost * pos.quantity + execute_price * quantity
            pos.quantity = total_shares
            pos.avg_cost = total_cost / total_shares
            pos.last_trade_date = current_date
            if self.config.t_plus_one:
                pos.available_quantity = pos.quantity - quantity  # T+1
            else:
                pos.available_quantity = total_shares
        else:
            pos = Position(
                ts_code=ts_code,
                quantity=quantity,
                avg_cost=execute_price,
                available_quantity=0 if self.config.t_plus_one else quantity,
                last_trade_date=current_date
            )
            positions[ts_code] = pos
        
        # 记录交易
        trade = Trade(
            trade_date=current_date,
            ts_code=ts_code,
            trade_type='buy',
            price=execute_price,
            quantity=quantity,
            amount=amount,
            commission=commission,
            stamp_duty=0,
            slippage=slippage,
            reason=signal_row.get('reason', '信号买入')
        )
        result.trades.append(trade)
        
        return cash
    
    def _execute_sell(
        self, signal_row: Dict, cash: float, positions: Dict[str, Position],
        price_dict: Dict, current_date: date, result: BacktestResult
    ) -> float:
        """执行卖出"""
        ts_code = signal_row['ts_code']
        
        if ts_code not in positions:
            return cash
        
        pos = positions[ts_code]
        if pos.available_quantity <= 0:
            return cash
        
        key = (ts_code, current_date)
        if key not in price_dict:
            return cash
        
        price_data = price_dict[key]
        pre_close = price_data['pre_close']
        limit_down = pre_close * (1 - self.config.price_limit)
        
        # 使用开盘价或限价
        execute_price = max(price_data['open'], limit_down)
        
        # 卖出全部可用
        quantity = pos.available_quantity
        
        # 计算费用
        amount = execute_price * quantity
        commission = max(amount * self.config.commission_rate, 5)
        stamp_duty = amount * self.config.stamp_duty_rate
        slippage = amount * self.config.slippage_rate
        total_revenue = amount - commission - stamp_duty - slippage
        
        cash += total_revenue
        
        # 更新持仓
        pos.quantity -= quantity
        pos.available_quantity -= quantity
        pos.last_trade_date = current_date
        
        if pos.quantity <= 0:
            del positions[ts_code]
        
        # 记录交易
        trade = Trade(
            trade_date=current_date,
            ts_code=ts_code,
            trade_type='sell',
            price=execute_price,
            quantity=quantity,
            amount=amount,
            commission=commission,
            stamp_duty=stamp_duty,
            slippage=slippage,
            reason=signal_row.get('reason', '信号卖出')
        )
        result.trades.append(trade)
        
        return cash
    
    def _normalize_benchmark(self, benchmark_data: pd.DataFrame, initial_value: float) -> pd.Series:
        """标准化基准数据"""
        first_value = benchmark_data['close'].iloc[0]
        normalized = (benchmark_data['close'] / first_value) * initial_value
        return pd.Series(normalized.values, index=benchmark_data['trade_date'])
    
    def _calculate_metrics(
        self, portfolio_values: pd.Series, benchmark_values: Optional[pd.Series]
    ) -> Dict:
        """计算回测指标"""
        if len(portfolio_values) < 2:
            return {}
        
        metrics = {}
        
        # 总收益率
        total_return = (portfolio_values.iloc[-1] / portfolio_values.iloc[0]) - 1
        metrics['total_return'] = total_return
        
        # 年化收益率
        n_days = len(portfolio_values)
        years = n_days / 252
        metrics['annual_return'] = (1 + total_return) ** (1 / years) - 1 if years > 0 else 0
        
        # 最大回撤
        cumulative = portfolio_values.values
        running_max = np.maximum.accumulate(cumulative)
        drawdown = (cumulative - running_max) / running_max
        metrics['max_drawdown'] = float(np.min(drawdown))
        
        # 夏普比率
        daily_returns = portfolio_values.pct_change().dropna()
        if len(daily_returns) > 0:
            risk_free_rate = 0.03  # 假设无风险利率3%
            excess_returns = daily_returns - (risk_free_rate / 252)
            if np.std(excess_returns) > 0:
                metrics['sharpe_ratio'] = float(np.mean(excess_returns) / np.std(excess_returns) * np.sqrt(252))
            else:
                metrics['sharpe_ratio'] = 0
        
        # 胜率（较基准）
        if benchmark_values is not None:
            common_dates = portfolio_values.index.intersection(benchmark_values.index)
            if len(common_dates) > 0:
                portfolio_rets = portfolio_values.loc[common_dates].pct_change().dropna()
                benchmark_rets = benchmark_values.loc[common_dates].pct_change().dropna()
                excess_rets = portfolio_rets - benchmark_rets
                win_rate = (excess_rets > 0).sum() / len(excess_rets)
                metrics['win_rate'] = float(win_rate)
        
        return metrics
```

```python
import re
import ast
from typing import Dict, List, Any, Optional
from dataclasses import dataclass

@dataclass
class IndicatorParam:
    """指标参数定义"""
    name: str
    param_type: str
    default: Any
    description: str = ""

@dataclass
class IndicatorMetadata:
    """指标元数据"""
    name: str
    description: str
    params: List[IndicatorParam]
    plots: List[str]
    has_signals: bool = False

class IndicatorContractParser:
    """指标协议解析器"""
    
    @staticmethod
    def parse(code: str) -> IndicatorMetadata:
        """解析指标代码"""
        metadata = IndicatorMetadata(
            name="未命名指标",
            description="",
            params=[],
            plots=[]
        )
        
        # 1. 提取名称
        name_match = re.search(r'my_strategy_name\s*=\s*["\']([^"\']+)["\']', code)
        if name_match:
            metadata.name = name_match.group(1)
        
        # 2. 提取描述
        desc_match = re.search(r'my_strategy_description\s*=\s*["\']([^"\']+)["\']', code)
        if desc_match:
            metadata.description = desc_match.group(1)
        
        # 3. 提取参数
        param_pattern = r'#\s*@param\s+(\w+)\s+(\w+)\s+([^\s#]+)\s*(.*)'
        params = re.findall(param_pattern, code)
        
        for name, param_type, default, description in params:
            parsed_default = IndicatorContractParser._parse_default(default, param_type)
            metadata.params.append(IndicatorParam(
                name=name,
                param_type=param_type,
                default=parsed_default,
                description=description.strip()
            ))
        
        # 4. 提取输出
        output_match = re.search(r'output\s*=\s*({[^}]+})', code, re.DOTALL)
        if output_match:
            try:
                output_str = output_match.group(1)
                # 简单解析 output
                plots_match = re.search(r'"plots"\s*:\s*\[([^\]]+)\]', output_str)
                if plots_match:
                    plots_str = plots_match.group(1)
                    plots = re.findall(r'"([^"]+)"', plots_str)
                    metadata.plots = plots
            except:
                pass
        
        # 5. 检查是否有信号
        if "signal" in code.lower():
            metadata.has_signals = True
        
        return metadata
    
    @staticmethod
    def _parse_default(value: str, param_type: str) -> Any:
        """解析默认值"""
        try:
            if param_type == 'int':
                return int(value)
            elif param_type == 'float':
                return float(value)
            elif param_type == 'bool':
                return value.lower() == 'true'
            else:
                return value
        except:
            return value
    
    @staticmethod
    def generate_param_config(params: List[IndicatorParam]) -> List[Dict]:
        """生成前端参数配置"""
        type_map = {
            'int': 'number',
            'float': 'number',
            'bool': 'switch',
            'str': 'input'
        }
        
        configs = []
        for param in params:
            configs.append({
                'name': param.name,
                'type': type_map.get(param.param_type, 'input'),
                'dataType': param.param_type,
                'default': param.default,
                'description': param.description
            })
        return configs
```

### 4.3 指标质量检查器

```python
import re
import ast
from typing import List, Dict, Any
from dataclasses import dataclass

@dataclass
class QualityIssue:
    """质量问题"""
    level: str  # error, warning, info
    message: str
    line: int = 0
    column: int = 0

class IndicatorQualityChecker:
    """指标质量检查器"""
    
    @staticmethod
    def check(code: str) -> List[QualityIssue]:
        """运行所有检查"""
        issues = []
        
        # 1. 检查是否有 df.copy()
        if not IndicatorQualityChecker._check_df_copy(code):
            issues.append(QualityIssue(
                level='warning',
                message='建议在修改DataFrame前使用 df = df.copy()，避免修改原始数据'
            ))
        
        # 2. 检查是否有未来函数
        future_funcs = IndicatorQualityChecker._check_future_functions(code)
        for func, line in future_funcs:
            issues.append(QualityIssue(
                level='error',
                message=f'检测到未来函数 {func}，在回测中会产生未来数据偏差',
                line=line
            ))
        
        # 3. 检查输出格式
        if not IndicatorQualityChecker._check_output(code):
            issues.append(QualityIssue(
                level='warning',
                message='未检测到标准的 output 字典定义'
            ))
        
        # 4. 检查危险操作
        danger_ops = IndicatorQualityChecker._check_dangerous_operations(code)
        for op, line in danger_ops:
            issues.append(QualityIssue(
                level='error',
                message=f'检测到危险操作 {op}，不允许在指标代码中使用',
                line=line
            ))
        
        # 5. 尝试解析语法
        try:
            ast.parse(code)
        except SyntaxError as e:
            issues.append(QualityIssue(
                level='error',
                message=f'语法错误: {e.msg}',
                line=e.lineno or 0,
                column=e.offset or 0
            ))
        
        return issues
    
    @staticmethod
    def _check_df_copy(code: str) -> bool:
        """检查是否有 df.copy()"""
        return bool(re.search(r'df\s*=\s*df\.copy', code))
    
    @staticmethod
    def _check_future_functions(code: str) -> List[tuple]:
        """检查未来函数"""
        issues = []
        lines = code.split('\n')
        
        # shift(-N) 模式
        shift_pattern = re.compile(r'\.shift\s*\(\s*-\s*\d')
        
        # iloc 未来数据模式
        iloc_future_pattern = re.compile(r'\.iloc\s*\[\s*[+-]')
        
        for i, line in enumerate(lines, 1):
            if shift_pattern.search(line):
                issues.append(('shift(-N)', i))
            if iloc_future_pattern.search(line):
                issues.append(('iloc[+N]', i))
        
        return issues
    
    @staticmethod
    def _check_output(code: str) -> bool:
        """检查输出格式"""
        return bool(re.search(r'output\s*=', code))
    
    @staticmethod
    def _check_dangerous_operations(code: str) -> List[tuple]:
        """检查危险操作"""
        danger_ops = [
            ('import os', 'import os'),
            ('import sys', 'import sys'),
            ('import subprocess', 'import subprocess'),
            ('open(', 'open('),
            ('eval(', 'eval('),
            ('exec(', 'exec('),
            ('__import__', '__import__'),
            ('requests.', 'requests.'),
            ('http', 'http'),
        ]
        
        issues = []
        lines = code.split('\n')
        
        for i, line in enumerate(lines, 1):
            for pattern, name in danger_ops:
                if pattern in line:
                    issues.append((name, i))
        
        return issues
```

### 4.4 指标沙盒执行器

**文件**: `backend/app/services/indicator_sandbox.py` (新建)

```python
import pandas as pd
import numpy as np
from typing import Dict, Any, Optional
import traceback
from io import StringIO

class IndicatorSandbox:
    """指标沙盒执行器"""
    
    # 允许的模块和函数
    ALLOWED_GLOBALS = {
        'pd': pd,
        'np': np,
        'pandas': pd,
        'numpy': np,
        'abs': abs,
        'max': max,
        'min': min,
        'sum': sum,
        'len': len,
        'range': range,
    }
    
    @staticmethod
    def execute(code: str, df: pd.DataFrame, params: Optional[Dict] = None) -> Dict[str, Any]:
        """
        在沙盒中执行指标代码
        
        参数:
            code: 指标代码
            df: 输入数据
            params: 参数字典
        
        返回:
            执行结果 output 字典
        """
        local_vars = {
            'df': df.copy(),  # 传入副本
        }
        
        # 设置参数
        if params:
            local_vars.update(params)
        
        try:
            # 执行代码
            exec(code, IndicatorSandbox.ALLOWED_GLOBALS, local_vars)
            
            # 获取输出
            output = local_vars.get('output', {})
            
            # 安全转换结果
            safe_output = IndicatorSandbox._make_safe(output)
            
            return {
                'success': True,
                'output': safe_output,
                'local_vars': {
                    k: IndicatorSandbox._make_safe(v)
                    for k, v in local_vars.items()
                    if not k.startswith('_')
                }
            }
        except Exception as e:
            return {
                'success': False,
                'error': str(e),
                'traceback': traceback.format_exc()
            }
    
    @staticmethod
    def _make_safe(obj: Any) -> Any:
        """将对象转换为安全可序列化的格式"""
        if isinstance(obj, pd.DataFrame):
            return obj.to_dict('records')
        if isinstance(obj, pd.Series):
            return obj.tolist()
        if isinstance(obj, np.ndarray):
            return obj.tolist()
        if isinstance(obj, (np.integer, np.int64, np.int32)):
            return int(obj)
        if isinstance(obj, (np.floating, np.float64, np.float32)):
            return float(obj)
        if isinstance(obj, dict):
            return {k: IndicatorSandbox._make_safe(v) for k, v in obj.items()}
        if isinstance(obj, (list, tuple)):
            return [IndicatorSandbox._make_safe(item) for item in obj]
        if pd.isna(obj) or (isinstance(obj, float) and np.isnan(obj)):
            return None
        return obj
```

### 5.3 研究流水线编排器

**文件**: `backend/app/services/research_pipeline.py` (新建)

```python
from typing import Dict, Any, Optional
from dataclasses import dataclass
from enum import Enum

class PipelineStatus(Enum):
    PENDING = "pending"
    RUNNING = "running"
    SUCCESS = "success"
    ERROR = "error"

@dataclass
class PipelineResult:
    """流水线结果"""
    status: PipelineStatus
    steps: Dict[str, Any]
    error: Optional[str] = None
    final_output: Optional[Dict] = None

class ResearchPipeline:
    """研究流水线编排器"""
    
    def __init__(self):
        self.steps = []
    
    def add_step(self, name: str, func):
        """添加步骤"""
        self.steps.append((name, func))
    
    def run(self, input_data: Dict) -> PipelineResult:
        """运行流水线"""
        result = PipelineResult(
            status=PipelineStatus.RUNNING,
            steps={},
            error=None
        )
        
        current_data = input_data
        
        for step_name, step_func in self.steps:
            try:
                step_output = step_func(current_data)
                result.steps[step_name] = {
                    'status': 'success',
                    'output': step_output
                }
                current_data = {**current_data, **step_output}
            except Exception as e:
                result.steps[step_name] = {
                    'status': 'error',
                    'error': str(e)
                }
                result.status = PipelineStatus.ERROR
                result.error = f"步骤 '{step_name}' 失败: {str(e)}"
                return result
        
        result.status = PipelineStatus.SUCCESS
        result.final_output = current_data
        return result
```
