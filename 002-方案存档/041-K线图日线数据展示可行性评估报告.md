```python
# kline_display.py

def display_kline(ts_code: str, period: str = 'daily'):
    """
    展示K线图
    
    Args:
        ts_code: 股票代码
        period: 周期（daily/weekly/monthly）
    """
    # 从缓存加载数据
    df = load_market_data(ts_code, period)
    
    # 计算复权价格
    adj_factor = load_adj_factor(ts_code)
    df['adj_close'] = df['close'] * adj_factor
    
    # 使用pyecharts或plotly绘制K线图
    kline = (
        Kline()
        .add_xaxis(df['trade_date'].tolist())
        .add_yaxis(
            "K线图",
            df[['open', 'close', 'low', 'high']].values.tolist()
        )
        .set_global_opts(
            title_opts=TitleOpts(title=f"{ts_code} K线图"),
            xaxis_opts=AxisOpts(type_="category"),
            yaxis_opts=AxisOpts(type_="value", min_="dataMin")
        )
    )
    
    return kline
```
