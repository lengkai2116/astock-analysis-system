```python
def get_minute_data(self, ts_code, freq='15min', start_date=None, end_date=None):
    """
    获取分钟线数据
    freq: 1min/5min/15min/30min/60min
    
    Tushare 5000积分档权限：
    - 1min: 当日数据
    - 5min/15min/30min/60min: 最近30天数据
    """
    ts.set_token(self.token)
    data = ts.pro_bar(
        ts_code=ts_code,
        start_date=start_date,
        end_date=end_date,
        freq=freq,
        adj='qfq'
    )
    return data.to_dict('records') if data is not None else []
```

### 4.2 应对方案

```python
# 请求频率控制示例
class RateLimiter:
    def __init__(self, max_calls=1, period=60):
        self.max_calls = max_calls
        self.period = period
        self.calls = []
    
    def allow(self):
        now = time.time()
        self.calls = [c for c in self.calls if now - c < self.period]
        if len(self.calls) < self.max_calls:
            self.calls.append(now)
            return True
        return False
```
